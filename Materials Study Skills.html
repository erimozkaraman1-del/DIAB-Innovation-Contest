function wrapLabels(label, maxWidth) {
const words = label.split(' ');
let lines = [];
let currentLine = '';
words.forEach(word => {
if ((currentLine + ' ' + word).trim().length > maxWidth) {
lines.push(currentLine);
currentLine = word;
} else {
currentLine = currentLine ? currentLine + ' ' + word :
word
;
}
});
if (currentLine) {
lines.push(currentLine);
}
return lines.length > 0 ? lines : [label];
}
const ctx =
document
.getElementById('projectTypesChart').getContext('2d');
const labels = [
'Physical Prototype',
'Working Model / Online Simulation',
'Animated Video'
];
const wrappedLabels = labels.map(label => wrapLabels(label, 16));
const projectTypesChart = new Chart(ctx, {
type: 'bar',
data: {

datasets: [{
label: 'Creativity Score',

'rgba(6, 214, 160, 0.7)'
],

'#FF6B6B',

'#06D6A0'
],
borderWidth: 2,

}]
},
options: {
responsive: true,

}
}

}
}

const ideaBtn = document.getElementById('generate-idea-btn');
const refineBtn = document.getElementById('refine-project-btn');
const ideaResponse = document.getElementById('idea-response');
const refineResponse = document.getElementById('refine-response');
const sdgSelect = document.getElementById('sdg-select');
const projectInput = document.getElementById('project-input');
const loadingSpinner = document.getElementById('loading-spinner');
async function fetchGeminiResponse(prompt, targetElement) {
loadingSpinner.classList.remove('hidden');
targetElement.textContent = '';
targetElement.classList.add('animate-pulse');
const apiKey = "";
const apiUrl =
`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-
preview-05-20:generateContent?key=
${apiKey}`;
try {
const payload = {
contents: [{ parts: [{ text: prompt }] }],
tools: [{ "google_search": {} }],
systemInstruction: {
parts: [{ text: "You are an expert innovation and
technology mentor for middle school students. Provide concise,
encouraging, and actionable advice. Use a friendly and supportive tone.
Keep your responses to a single, well-structured paragraph." }]
},
};
const response = await fetch(apiUrl, {
method: 'POST',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify(payload)
});
if (!response.ok) {
throw new Error(`HTTP error! status:
${response.status}`);
}
const result = await response.json();
const text =
result
?.candidates?.[0]?.content?.parts?.[0]?.text || "Sorry, I couldn't
generate a response. Please try again."
;
targetElement.textContent = text;
} catch (error) {
console.error('Fetch error:', error);
targetElement.textContent = `Error: ${error.message}.
Please try again.`
;
} finally {
loadingSpinner.classList.add('hidden');
targetElement.classList.remove('animate-pulse');
}
}
ideaBtn.addEventListener('click', () => {
const selectedSdg = sdgSelect.value;

Contest focused on "
${selectedSdg}". The project can be a physical
prototype, a working model/online simulation, or an animated video. The
idea should be suitable for middle school students.`
;
fetchGeminiResponse(prompt, ideaResponse);
});
refineBtn.addEventListener('click', () => {
const projectIdea = projectInput.value.trim();
if (!projectIdea) {
refineResponse.textContent = "Please enter a project idea
to get feedback."
;
return;
}
const prompt = `I am a student with a project idea for an
innovation contest. My idea is: "
${projectIdea}". As an expert mentor,
please provide me with a few constructive suggestions to improve it, and
tell me a good next step.`
;
fetchGeminiResponse(prompt, refineResponse);
});
